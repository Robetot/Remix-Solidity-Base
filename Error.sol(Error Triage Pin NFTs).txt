pragma solidity ^0.8.20;

contract ErrorTriageExercise {
    /**
     * Finds the difference between each uint with its neighbor (a to b, b to c, etc.)
     * and returns a uint array with the absolute integer difference of each pairing.
     */
    function diffWithNeighbor(
        uint _a,
        uint _b,
        uint _c,
        uint _d
    ) public pure returns (uint[] memory) {
        uint[] memory results = new uint[](3);

        results[0] = _a > _b ? _a - _b : _b - _a;
        results[1] = _b > _c ? _b - _c : _c - _b;
        results[2] = _c > _d ? _c - _d : _d - _c;

        return results;
    }

    /**
     * Changes the _base by the value of _modifier.
     * Base is always >= 1000. Modifiers can be between -100 and +100.
     */
    function applyModifier(
        uint _base,
        int _modifier
    ) public pure returns (uint) {
        require(_base >= 1000, "Base must be >= 1000");

        if (_modifier >= 0) {
            return _base + uint(_modifier);
        } else {
            // Convert negative int to positive uint safely
            uint absModifier = uint(-_modifier);
            require(absModifier <= 100, "Modifier too large");
            require(_base >= absModifier, "Underflow: result would be < 0");
            return _base - absModifier;
        }
    }

    /**
     * Pop the last element from the supplied array, and return the popped value.
     */
    uint[] private arr; // Made private for encapsulation

    function popWithReturn() public returns (uint) {
        require(arr.length > 0, "Array is empty");

        uint lastElement = arr[arr.length - 1];
        arr.pop(); // Removes last element
        return lastElement;
    }

    // Utility functions
    function addToArr(uint _num) public {
        arr.push(_num);
    }

    function getArr() public view returns (uint[] memory) {
        return arr;
    }

    function resetArr() public {
        delete arr;
    }
}
